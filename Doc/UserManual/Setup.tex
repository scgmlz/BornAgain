\def\@part[#1]#2{%
    \refstepcounter{part}%
    \addcontentsline{toc}{part}{Part~\thepart\hspace{1em}#1}%
    \markboth{}{}%
    {\interlinepenalty \@M
     \normalfont
     \ifnum \c@secnumdepth >-2\relax
       \parindent \z@ \Large\bfseries \partname\nobreakspace\thepart
       \par
       \vskip 20\p@
     \fi
     \parindent \z@ \huge \bfseries #2\par}
    }
